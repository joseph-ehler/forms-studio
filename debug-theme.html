Run this in your browser console to check theme CSS variables:

// PASTE THIS IN CONSOLE:

function debugThemeVariables() {
  const root = document.documentElement;
  const brand = root.dataset.brand || 'default';
  const theme = root.dataset.theme || 'light';
  const computedStyle = getComputedStyle(root);
  
  console.group(`üé® Theme Debug: brand="${brand}", theme="${theme}"`);
  
  // Critical variables that MUST be defined
  const criticalVars = [
    '--ds-color-text-primary',
    '--ds-color-text-secondary',
    '--ds-color-text-muted',
    '--ds-color-surface-base',
    '--ds-color-surface-subtle',
    '--ds-color-surface-raised',
    '--ds-color-border-subtle',
    '--ds-color-border-strong',
    '--ds-color-primary-bg',
    '--ds-color-primary-text',
  ];
  
  console.log('Checking critical CSS variables:');
  
  let missingCount = 0;
  criticalVars.forEach(varName => {
    const value = computedStyle.getPropertyValue(varName).trim();
    
    if (!value || value === '') {
      console.error(`‚ùå MISSING: ${varName}`);
      missingCount++;
    } else {
      console.log(`‚úÖ ${varName}: ${value}`);
    }
  });
  
  if (missingCount > 0) {
    console.error(`\nüö® PROBLEM FOUND: ${missingCount} critical variables are missing!`);
    console.log(`\nThis means the theme="${theme}" + brand="${brand}" combination is incomplete.`);
    console.log(`\nExpected behavior:`);
    console.log(`- Default brand: Has full dark mode styles ‚úÖ`);
    console.log(`- Other brands: Only override brand colors, missing neutrals ‚ùå`);
  } else {
    console.log(`\n‚úÖ All critical variables are defined!`);
  }
  
  console.groupEnd();
}

// Test all combinations:
console.log('üß™ Testing all brand √ó theme combinations:\n');

['default', 'acme', 'techcorp', 'sunset'].forEach(brand => {
  ['light', 'dark'].forEach(theme => {
    document.documentElement.dataset.brand = brand;
    document.documentElement.dataset.theme = theme;
    debugThemeVariables();
  });
});
